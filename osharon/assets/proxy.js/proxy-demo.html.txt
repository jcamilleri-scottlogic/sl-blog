    <script>
        function Context(initial = {}) {
  const listeners = new Map();
  const computed = new Map();
  const notify = (key, value) => {
    if (listeners.has(key)) {
      listeners.get(key).forEach(fn => fn(value));
    }
  };

  const handler = {
    get(target, prop, receiver) {
      if (computed.has(prop)) {
        return computed.get(prop)(receiver); // compute dynamically
      }
      return Reflect.get(...arguments);
    },
    set(target, prop, value, receiver) {
      const result = Reflect.set(...arguments);
      notify(prop, value);
      return result;
    },
    delete(target, prop) {
      const result = Reflect.deleteProperty(...arguments);
        if (computed.has(prop)) {
          computed.delete(prop);
        }
        return result;
      }
    };

  const proxy = new Proxy(initial, handler);

  proxy.addListener = (prop, fn) => {
    if (!listeners.has(prop)) {
      listeners.set(prop, new Set());
    }
    listeners.get(prop).add(fn);
  };

  proxy.removeListener = (prop, fn) => {
    if (listeners.has(prop)) {
      listeners.get(prop).delete(fn);
    }
  };

  proxy.compute = (prop, fn) => {
    return computed.set(prop, fn);;
  };

  proxy.bidi = (prop, elm, attribute = 'value', event = 'input') => {
    elm[attribute] = proxy[prop] || '';
    elm.addEventListener(event, () => proxy[prop] = elm[attribute]);
    proxy.addListener(prop, (newValue) => {
      if (elm[attribute] !== newValue) {
        elm[attribute] = newValue || '';
      }
    });
  };

  return proxy;
}

<script type="module">
    document.addEventListener('DOMContentLoaded',() => {
        const state = new Context({ count: 0 });

        state.addListener('count', (newValue) => {
            document.getElementById('countDisplay').textContent = newValue;
            const notification = document.getElementById('notification');
            
            if (newValue > 10) {
                state.name = '';
                notification.hidden = false;
            } else {
                notification.hidden = true;
            }
        });

        state.bidi('name', document.getElementById('name'));
        state.addListener('name', (newValue) => {
            document.getElementById('welcomeMessage').textContent = newValue.length > 0 ? `Welcome, ${newValue}!` : 'Enter your name';
        });
  
      window.state = state; // exposing it to code within the html;
    });
</script>
